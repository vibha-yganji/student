def draw_branch(branch_length, t):
    if branch_length > 5:
        # Draw the main branch
        t.forward(branch_length)
        t.right(20)
        draw_branch(branch_length - 15, t)
        
        # Draw the right sub-branch (recursion)
        t.left(40)
        draw_branch(branch_length - 15, t)
        
        # Return to the original orientation and position
        t.right(20)
        t.backward(branch_length)

def main():
    # Setup the Turtle
    window = turtle.Screen()
    window.bgcolor("white")
    
    t = turtle.Turtle()
    t.color("brown")
    t.width(2)
    t.left(90)  # Start by pointing up
    
    # Set the starting position
    t.penup()
    t.goto(0, -200)
    t.pendown()
    
    # Initial branch length
    branch_length = 100
    
    # Call the recursive function to draw the tree
    draw_branch(branch_length, t)
    
    # Close the window on click
    window.exitonclick()

if __name__ == "__main__":
    main()

This code creates a simple recursive tree using the Turtle graphics library. The draw_branch function is called recursively to draw the tree branches. The main function sets up the Turtle environment, sets the initial position, and starts the drawing. When you click the Turtle graphics window, it will close. You can modify the branch length, angle, and other parameters to create different tree shapes or experiment with other recursive patterns.
